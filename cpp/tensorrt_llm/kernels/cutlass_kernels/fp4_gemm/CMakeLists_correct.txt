cmake_minimum_required(VERSION 3.20)
project(CorrectFp4GemmTest)

# 设置 C++ 标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 查找 CUDA
find_package(CUDA REQUIRED)
enable_language(CUDA)

# 设置 CUDA 架构
set(CMAKE_CUDA_ARCHITECTURES "80;86;89;90")

# 创建可执行文件
add_executable(correct_fp4_gemm_test ../correct_fp4_gemm_test.cpp)

# 设置包含目录
target_include_directories(correct_fp4_gemm_test PRIVATE
    ${CUDA_INCLUDE_DIRS}
)

# 链接库目录
link_directories(${CUDA_TOOLKIT_ROOT_DIR}/lib64)

# 链接库
target_link_libraries(correct_fp4_gemm_test
    ${CUDA_LIBRARIES}
    cublasLt
)

# 设置编译选项
target_compile_options(correct_fp4_gemm_test PRIVATE
    -O2
    -Wall
    -Wextra
)

# 设置 CUDA 编译选项
set_target_properties(correct_fp4_gemm_test PROPERTIES
    CUDA_SEPARABLE_COMPILATION ON
)
